Question 1: You are given a string (STR) of length N, consisting of only the lower case English alphabet. Your task is to remove all the duplicate occurrences of characters in the string.

Input: abcadeecfb
Output: abcdef 

        Code

        const set = new Set();
        function duplicate(str,n)
        {
            for(let i=0;i < str.length;i++)
            {
                set.add(str[i]);
            }
            let str1 = " ";
            for(let val of set .values())
            {
                str1 = str1+val;
            }

Question 2: You are given a string (STR) of length N, you have to print the count of all alphabet.(using maps)

Input: abcadeecfb

Output:
a=2
b=2
c=2
d=1
e=2
f=1

        Code

        let str = "abcdeecfb";
        let newStr = str.split("");
        let newMap = new Map();
        
        for (let i = 0; i < newStr.length; i++) {
            let count = 1;
            for (let b = i + 1; b < newStr.length; b++) {
                if (newStr[i] === newStr[b]) {
                    count++;
                    newStr.splice(b, 1);
                    b--;
                }
            }
            newMap.set(newStr[i], count);
        }
        for (let [key, value] of newMap.entries()) {
            console.log( key + "=" + value);
        }
